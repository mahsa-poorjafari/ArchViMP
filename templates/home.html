            void diff() {

            pthread_mutex_lock(&mutex2);
            sharedVar2++;
            pthread_mutex_unlock(&mutex2);
            int a=5;
            int b=5;
            b=a+b;
            pthread_mutex_lock(&mutex2);
            result = a-b;
            pthread_mutex_unlock(&mutex2);

            sharedVar2++;

            pthread_mutex_lock(&mutex2);
            sharedVar2++;
            pthread_mutex_unlock(&mutex2);

            sharedVar1++;

            //cout << "result from 2nd thread is" << result <<endl;
            //pthread_exit(NULL);
            }


            void incrementSharedVars() {

            pthread_mutex_lock(&mutex2);
            sharedVar2++;
            pthread_mutex_unlock(&mutex2);
            sharedVar1;
            }


            int main () {
            pthread_t thread1 , thread2, thread3, thread4;

            int rc;
